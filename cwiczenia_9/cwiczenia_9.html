# Importowanie biblioteki matplotlib:
import matplotlib.pyplot as plt

# Generowanie x i y losowych punktów:
x = np.random.uniform(0.0, 10.0, 16)
y = np.random.uniform(0.0, 10.0, 16)

# Tworzenie wykresu punktowego:
plt.scatter(x, y, c="red", s=10)

# Wyświetlenie wykresu:
plt.show()

# Tworzenie 1000 punktów z przedziału od 0.0 do 8.0:
x = np.linspace(0.0, 8.0, 1000)

# Wyznaczenie wartości sin dla każdego punktu:
y = np.sin(x)

# Rysowanie wykresu ciągłego:
plt.plot(x, y, color="green")
plt.show()

Dalsze przykłady:¶

In [33]:
# Generowanie domeny x
x = np.linspace(0.0, 8.0, 1000)

# Wyznaczanie wartości funkcji:
f = 5 * np.sin(x)
g = 5 * np.cos(x)
h = (x - 1) * (x - 4) * (x - 7)

# Rysowanie wszystkich funkcji:
plt.plot(x, f, color="red", label="f(x)")
plt.plot(x, g, color="green", label="g(x)")
plt.plot(x, h, color="cyan", label="h(x)")

# Wyświetlenie legendy:
plt.legend()

# Nadanie tytułu wykresu:
plt.title("Wykresy funkcji")

# Podpis osi X:
plt.xlabel("Wartości X")

# Podpis osi Y:
plt.ylabel("Wartości Y")

# Wyświetlenie siatki:
plt.grid()

plt.show()
In [34]:
# Generowanie punktów x i y (rozkład normalny):
x_1 = np.random.normal(0.0, 1.0, 100)
y_1 = np.random.normal(0.0, 1.0, 100)

# Generowanie punktów x i y (rozkład jednostajny):
x_2 = np.random.uniform(-1.0, 1.0, 100)
y_2 = np.random.uniform(-1.0, 1.0, 100)

# Tworzenie wykresu składającego się z dwóch podwykresów:
fig, (ax1, ax2) = plt.subplots(1, 2)

# Ustalenie wielkość wykresu:
fig.set_size_inches(8, 4)

# Wypełnienie pierwszego podwykresu:
ax1.scatter(x_1, y_1, c="red", s=10)
ax1.set_xlim(-5, 5)
ax1.set_ylim(-5, 5)
ax1.set_title("Rozkład normalny")
ax1.set_xlabel("X")
ax1.set_ylabel("Y")

# Wypełnienie drugiego podwykresu:
ax2.scatter(x_2, y_2, c="green", s=10)
ax2.set_xlim(-5, 5)
ax2.set_ylim(-5, 5)
ax2.set_title("Rozkład jednostajny")
ax2.set_xlabel("X")
ax2.set_ylabel("Y")

# Zapisanie wykresu do pliku .png:
plt.savefig("wykres.png", dpi=320)

# Wyświetlenie wykresu:
plt.show()
