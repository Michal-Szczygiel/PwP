# Importowanie biblioteki matplotlib:
import matplotlib.pyplot as plt

# Generowanie x i y losowych punktw:
x = np.random.uniform(0.0, 10.0, 16)
y = np.random.uniform(0.0, 10.0, 16)

# Tworzenie wykresu punktowego:
plt.scatter(x, y, c="red", s=10)

# Wywietlenie wykresu:
plt.show()

Dalsze przykady:

In[33]:
# Generowanie domeny x
x = np.linspace(0.0, 8.0, 1000)

# Wyznaczanie wartoci funkcji:
f = 5 * np.sin(x)
g = 5 * np.cos(x)
h = (x - 1) * (x - 4) * (x - 7)

# Rysowanie wszystkich funkcji:
plt.plot(x, f, color="red", label="f(x)")
plt.plot(x, g, color="green", label="g(x)")
plt.plot(x, h, color="cyan", label="h(x)")

# Wywietlenie legendy:
plt.legend()

# Nadanie tytuu wykresu:
plt.title("Wykresy funkcji")

# Podpis osi X:
plt.xlabel("Wartoci X")

# Podpis osi Y:
plt.ylabel("Wartoci Y")

# Wywietlenie siatki:
plt.grid()

plt.show()
In[34]:
# Generowanie punktw x i y (rozkad normalny):
x_1 = np.random.normal(0.0, 1.0, 100)
y_1 = np.random.normal(0.0, 1.0, 100)

# Generowanie punktw x i y (rozkad jednostajny):
x_2 = np.random.uniform(-1.0, 1.0, 100)
y_2 = np.random.uniform(-1.0, 1.0, 100)

# Tworzenie wykresu skadajcego si z dwch podwykresw:
fig, (ax1, ax2) = plt.subplots(1, 2)

# Ustalenie wielko wykresu:
fig.set_size_inches(8, 4)

# Wypenienie pierwszego podwykresu:
ax1.scatter(x_1, y_1, c="red", s=10)
ax1.set_xlim(-5, 5)
ax1.set_ylim(-5, 5)
ax1.set_title("Rozkad normalny")
ax1.set_xlabel("X")
ax1.set_ylabel("Y")

# Wypenienie drugiego podwykresu:
ax2.scatter(x_2, y_2, c="green", s=10)
ax2.set_xlim(-5, 5)
ax2.set_ylim(-5, 5)
ax2.set_title("Rozkad jednostajny")
ax2.set_xlabel("X")
ax2.set_ylabel("Y")

# Zapisanie wykresu do pliku .png:
plt.savefig("wykres.png", dpi=320)

# Wywietlenie wykresu:
plt.show()
